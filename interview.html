const Dashboard = () => {
  const [ready, setReady] = useState(false);
  const [error, setError] = useState(null);

  try {
    useEffect(() => {
      initDashboard();  // throws if config is invalid
      setReady(true);
    }, []);
  } catch (e) {
    setError(e.message);
  }

  if (error) return <div>Something went wrong: {error}</div>;
  if (!ready) return <div>Initializing...</div>;
  return <DashboardContent />;
};

Settings.jsx 
const Settings = ({ rawConfig }) => {
  const [hasError, setHasError] = useState(false);

  try {
    const config = parseConfig(rawConfig); // throws if rawConfig is malformed
  } catch (e) {
    setHasError(true);
  }

  if (hasError) return <div>Invalid configuration</div>;
  return <SettingsPanel config={config} />;
};
DCh 16 >UserProfile.jsx
UserProfile.jsx 
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    fetch(`/api/users/${userId}`)
      .then(r => {
        if (!r.ok) throw new Error('Failed to fetch');
        return r.json();
      })
      .then(data => setUser(data));
  }, [userId]);

  return (
    <ErrorBoundary>
      <div>{user?.name}</div>
    </ErrorBoundary>
  );
}
